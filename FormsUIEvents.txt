🔹 1. Understanding Forms in React
In React, forms are used to collect user input. Unlike HTML, React uses controlled components (state-driven).
Input values are controlled using state + onChange

✅ Example
```jsx
import React, { useState } from "react";

function LoginForm() {
  const [username, setUsername] = useState("");

  return (
    <form>
      <input type="text" value={username}  
        onChange={(e) => setUsername(e.target.value)}/>
      <p>Entered: {username}</p>
    </form>
  );
}export default LoginForm;

🔹 6. Multiple Checkboxes (Real-time)

 ✅ Example
const [items,SetItems]=useState(["A","B","C"]);
setItems([...items,"C"]);
output:
["A","B","C"]

```jsx
function MultiCheckbox() {
  const [skills, setSkills] = useState([]);

  function handleChange(e) {
    const value = e.target.value;

    if (e.target.checked) {
      setSkills([...skills, value]);  //spread operator
    } else {
      setSkills(skills.filter((item) => item !== value));
    }
  }

  return (
    <div>
      <input type="checkbox" value="React" onChange={handleChange} /> React
      <input type="checkbox" value="Python" onChange={handleChange} /> Python

      <p>Skills: {skills.join(", ")}</p>
    </div>
  );
}

🔹 3. Handling User Inputs
 Use `onChange` event to update state.
Handling user inputs in React is mainly about capturing what the user types and storing it in state so your UI updates dynamically.

🔹 What is Handling User Input?

👉 When a user types, selects, or clicks something → React captures it using events and updates state

✔ `useState` → stores data
✔ `onChange` → captures input

---

🔹 1. Basic Example (Single Input)

✅ Example

```jsx
import React, { useState } from "react";

function InputExample() {
  const [name, setName] = useState("");

  function handleChange(e) {
    setName(e.target.value);
  }

  return (
    <div>
      <input type="text"
        placeholder="Enter name"
        value={name}
        onChange={handleChange}
      />
      <p>You typed: {name}</p>
    </div>
  );
}

export default InputExample;


 Flow:

* User types → `onChange` triggers
* `handleChange` updates state
* UI re-renders

🔹 2. Handling Multiple Inputs
 Use **single function** for all inputs

 ✅ Example

```jsx
import React, { useState } from "react";

function FormExample() {
  const [form, setForm] = useState({
    name: "",
    email: ""
  });

  function handleChange(e) {
    const { name, value } = e.target;

    setForm({
      ...form,
      [name]: value
    });
  }

  return (
    <div>
      <input
        name="name"
        placeholder="Name"
        onChange={handleChange}
      />

      <input
        name="email"
        placeholder="Email"
        onChange={handleChange}
      />

      <p>Name: {form.name}</p>
      <p>Email: {form.email}</p>
    </div>
  );
}

 Key idea:

* `name` attribute helps identify field
* `[name]: value` updates correct field

🔹 5. Understanding Form Validations
Form validation ensures that user input is correct before submitting

 ✅ Examples:

* Email must contain `@`
* Password must be at least 6 characters
* Required fields should not be empty

🔹 Types of Validation

 1. Client-side validation (React)

 Done in browser using JavaScript

 2. Server-side validation
 Done in backend (API)

In React, we mainly focus on **client-side validation**


🔹 1. Basic Validation Example

 ✅ Check Empty Field

```jsx
import React, { useState } from "react";

function SimpleForm() {
  const [name, setName] = useState("");
  const [error, setError] = useState("");

  function handleSubmit(e) {
    e.preventDefault();

    if (name === "") {
      setError("Name is required");
    } else {
      setError("");
      alert("Form submitted");
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <input
        placeholder="Enter name"
        onChange={(e) => setName(e.target.value)}
      />
      <p style={{ color: "red" }}>{error}</p>
      <button>Submit</button>
    </form>
  );
}

🔹 4. Multiple Field Validation (Real-Time Example)

✅ Login Form Validation

```jsx
function LoginForm() {
  const [data, setData] = useState({
    email: "",
    password: ""
  });

  const [errors, setErrors] = useState({});

  function handleChange(e) {
    setData({ ...data,
      [e.target.name]: e.target.value
    });
  }

  function validate() {
    let tempErrors = {};

    if (!data.email.includes("@")) {
      tempErrors.email = "Invalid email";
    }

    if (data.password.length < 6) {
      tempErrors.password = "Password too short";
    }

    return tempErrors;
  }

  function handleSubmit(e) {
    e.preventDefault();

    const validationErrors = validate();

    if (Object.keys(validationErrors).length > 0) {
      setErrors(validationErrors);
    } else {
      alert("Login successful");
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <input name="email" onChange={handleChange} />
      <p style={{ color: "red" }}>{errors.email}</p>

      <input type="password" name="password" onChange={handleChange} />
      <p style={{ color: "red" }}>{errors.password}</p>

      <button>Login</button>
    </form>
  );
}
